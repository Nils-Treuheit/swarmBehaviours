397 static float getDistance(struct LlaCoor_f *own_pos, struct LlaCoor_f *goal_pos)
398 {
399  return GLOBE_RADIUS * acos(
400  sin(own_pos->lat) * sin(goal_pos->lat) +
401  cos(own_pos->lat) * cos(goal_pos->lat) *
402  cos(own_pos->lon - goal_pos->lon));
403 }

435 static struct EnuCoor_f attract(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float multiplier)
436 {
437  struct EnuCoor_f force = {
438  pos_ac->x - own_pos->x,
439  pos_ac->y - own_pos->y,
440  0.0f};
441  msg.attraction_force = force;
442 
443  float d = sqrtf(force.x * force.x + force.y * force.y);
444  msg.attraction_d = d;
445  msg.attraction = true;
446 
447  d = fmaxf(0.0001f, d);
448  float strength = GRAVITY * multiplier / d;
449  msg.attraction_strength = strength;
450 
451  force.x = force.x * strength;
452  force.y = force.y * strength;
453  msg.attraction_force = force;
454 
455  acc->x += force.x;
456  acc->y += force.y;
457 
458  return force;
459 }

465 static struct EnuCoor_f repulse(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float perlimiter, float multiplier)
466 {
467  struct EnuCoor_f force = {
468  pos_ac->x - own_pos->x,
469  pos_ac->y - own_pos->y,
470  0.0f};
471  msg.repulsion_force = force;
472 
473  float d = sqrtf(force.x * force.x + force.y * force.y);
474  msg.repulsion_d = d;
475  msg.repulsion = true;
476 
477  float strength = (GRAVITY * multiplier) * expf(-(d * d) / (2 * perlimiter * perlimiter));
478  msg.repulsion_strength = strength;
479 
480  force.x = force.x * strength;
481  force.y = force.y * strength;
482  msg.repulsion_force = force;
483 
484  acc->x -= force.x;
485  acc->y -= force.y;
486 
487  return force;
488 }

494 static struct EnuCoor_f attRep(struct EnuCoor_f *own_pos, struct EnuCoor_f *pos_ac, struct EnuCoor_f *acc, float reg_size, float comfy_dist, float att_multiplier, float rep_multiplier)
495 {
496  struct EnuCoor_f force = {
497  pos_ac->x - own_pos->x,
498  pos_ac->y - own_pos->y,
499  0.0f};
500  msg.repulsion_force = force;
501  msg.attraction_force = force;
502 
503  float d = sqrtf(force.x * force.x + force.y * force.y);
504  float c = (reg_size * reg_size) / logf(rep_multiplier / att_multiplier);
505  msg.repulsion_d = d;
506  msg.attraction_d = d;
507  msg.repulsion = true;
508  msg.attraction = true;
509 
510  float strength_att = (GRAVITY * att_multiplier) * (fmax(d - comfy_dist, 0.0f) / fmax(d, 1.0f));
511  float strength_rep = (GRAVITY * rep_multiplier) * expf(-(d * d) / c);
512  msg.attraction_strength = strength_att;
513  msg.repulsion_strength = strength_rep;
514 
515  force.x = force.x * (strength_att - strength_rep);
516  force.y = force.y * (strength_att - strength_rep);
517  msg.attraction_force = force;
518  msg.repulsion_force = force;
519 
520  acc->x += force.x;
521  acc->y += force.y;
522 
523  return force;
524 }

530 void swarm_follow_wp(void)
531 {
532  struct EnuCoor_f att_force = {0.0f, 0.0f, 0.0f};
533  struct EnuCoor_f member_force = {0.0f, 0.0f, 0.0f};
534  struct EnuCoor_f rep_force = {0.0f, 0.0f, 0.0f};
535  setPoint(&acc, 0.0f, 0.0f, 0.0f);
536 
537  struct EnuCoor_f *own_pos = stateGetPositionEnu_f();
538  msg.own_pos = *own_pos;
539 
540  for (uint8_t ac_id = FIRST_SWARM_MEMBER_ID; ac_id <= LAST_SWARM_MEMBER_ID; ++ac_id)
541  {
542  struct EnuCoor_f *ac_pos = getPositionEnu_f(ac_id);
543  if (ac_pos != NULL && ac_id != AC_ID)
544  {
545  msg.target_pos = *ac_pos;
546  msg.target_ac_id = ac_id;
547  addToPoint(&member_force, attRep(own_pos, ac_pos, &acc, DRONE_REGION_SIZE, COMFY_DIST, DRONE_ATTRECTION_MULTIPLIER, DRONE_REPULSION_MULTIPLIER));
548  }
549  else
550  {
551  msg.target_ac_id = 0;
552  msg.attraction = false;
553  msg.repulsion = false;
554  }
555  }
556 
557  for (uint8_t wp_id = FIRST_ATTRACTION_POINT_ID; wp_id < LAST_ATTRACTION_POINT_ID; ++wp_id)
558  {
559  struct EnuCoor_f current_att_point = {0.0f, 0.0f, 0.0f};
560  current_att_point.x = waypoint_get_x(wp_id);
561  current_att_point.y = waypoint_get_y(wp_id);
562  if (current_att_point.x != 0.0f && current_att_point.y != 0.0f)
563  addToPoint(&att_force, attract(own_pos, &current_att_point, &acc, ATTRECTION_MULTIPLIER));
564  }
565 
566  for (uint8_t wp_id = FIRST_REPELL_POINT_ID; wp_id < LAST_REPELL_POINT_ID; ++wp_id)
567  {
568  struct EnuCoor_f current_rep_point = {0.0f, 0.0f, 0.0f};
569  current_rep_point.x = waypoint_get_x(wp_id);
570  current_rep_point.y = waypoint_get_y(wp_id);
571  if (current_rep_point.x != 0.0f && current_rep_point.y != 0.0f)
572  addToPoint(&rep_force, repulse(own_pos, &current_rep_point, &acc, REGION_SIZE, REPULSION_MULTIPLIER));
573  }
574 
575  struct EnuCoor_f *vel = acInfoGetVelocityEnu_f(AC_ID);
576  vel->x = VELOCITY_LIMIT * tanhf(vel->x + acc.x);
577  vel->y = VELOCITY_LIMIT * tanhf(vel->y + acc.y);
578  acInfoSetVelocityEnu_f(AC_ID, vel);
579 
580  struct EnuCoor_i future_pos = *stateGetPositionEnu_i();
581  future_pos.x += POS_BFP_OF_REAL(vel->x);
582  future_pos.y += POS_BFP_OF_REAL(vel->y);
583  future_pos.z = POS_BFP_OF_REAL(FLIGHT_HEIGHT);
584 
585  if (ticks == 0)
586  {
587  ticks = 3;
588  write_to_log_file(&future_pos, vel, &acc);
589  write_to_debug_file(&future_pos, vel, &acc, &att_force, &member_force, &rep_force);
590  }
591  else
592  --ticks;
593 
594  // Move the waypoints
595  current_pos = *stateGetPositionLla_f();
596  waypoint_set_enu_i(SWARM_WAYPOINT_ID, &future_pos);
597  for (uint8_t wp_id = FIRST_ATTRACTION_POINT_ID; wp_id < LAST_ATTRACTION_POINT_ID; ++wp_id)
598  {
599  struct LlaCoor_i *way_point = waypoint_get_lla(wp_id);
600  att_point = toFloatPointFormat(way_point);
601  updateSyncLinkMsg(&current_pos, wp_id);
602  }
603 }
