31 # gets the metric distance between two points given in the LlaCoor_f format
32 def getDistance(own_pos_lat, own_pos_lon, goal_pos_lat, goal_pos_lon)->float:
33  # GLOBE_RADIUS:int = int(6371000)
34  own_pos_lat *= math.pi/180
35  own_pos_lon *= math.pi/180
36  goal_pos_lat *= math.pi/180
37  goal_pos_lon *= math.pi/180
38  return float( 6371000 * math.acos(
39  math.sin(own_pos_lat) * math.sin(goal_pos_lat) +
40  math.cos(own_pos_lat) * math.cos(goal_pos_lat) *
41  math.cos(own_pos_lon - goal_pos_lon)
42  ))

44 # func to calc whiskers for violinplots
45 def adjacent_values(vals, q1, q3):
46  upper_adjacent_value = q3 + (q3 - q1) * 1.5
47  upper_adjacent_value = np.clip(upper_adjacent_value, q3, vals[-1])
48  lower_adjacent_value = q1 - (q3 - q1) * 1.5
49  lower_adjacent_value = np.clip(lower_adjacent_value, vals[0], q1)
50  return lower_adjacent_value, upper_adjacent_value
