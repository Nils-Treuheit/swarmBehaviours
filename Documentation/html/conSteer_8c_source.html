191 static Point point_from_angle(float radians)

194 static Point sub_points(Point* a, Point* b)
195 { return init_point((a->x - b->x), (a->y - b->y)); }
196 
197 static Point add_points(Point* a, Point* b)
198 { return init_point((a->x + b->x), (a->y + b->y)); }
199 
200 static Point scale_up_points(Point* a, float s)
201 { return init_point((a->x * s), (a->y * s)); }
202 
203 static Point scale_down_points(Point* a, float s)
204 { return init_point((a->x / s), (a->y / s)); }

456 //calculate repulsion forces to other drones
457 static Point linear_Repulsion(Point* pos, Point* target, float max_val)
458 {
459  Point force = sub_points(target, pos);
460  float magnitude = mag(&force);
461  float d = magnitude; //fminf(magnitude, limit);
462  float strength = (-1)*max_val*GRAVITY/fmaxf(1.0f,d);
463  force = scale_up_points(&force,(strength/magnitude));
464  return force;
465 }

467 // calculate attraction forces to other drones
468 static Point log_Attraction(Point* pos, Point* target, float cutOff)
469 {
470  Point force = sub_points(target, pos);
471  float magnitude = mag(&force);
472  float d = magnitude; //fminf(magnitude, limit);
473  float strength = 0;
474  if(d>cutOff) strength = log(1+(d-cutOff))*GRAVITY;
475  force = scale_up_points(&force,(strength/magnitude));
476  return force;
477 }

479 // calculate attraction forces with GOAL_VECTORS
480 static Point linear_Attraction(Point* pos, Point* target, float max_val)
481 {
482  Point force = sub_points(target, pos);
483  float magnitude = mag(&force);
484  float d = magnitude; //fminf(magnitude, limit);
485  float strength = (GRAVITY * max_val)/fmaxf(d,1.0f);
486  force = scale_up_points(&force,(strength/magnitude));
487  return force;
488 }

490 // calculate repulsion force with DANGER_VECTORS
491 static Point limExp_Repulsion(Point* pos, Point* target, float cutOff, float max_val)
492 {
493  Point force = sub_points(target, pos);
494  float magnitude = mag(&force);
495  float strength;
496  float d = magnitude; //fminf(magnitude, limit);
497  if (d<cutOff) strength = (-1)*(GRAVITY * max_val)/fmaxf(d,1.0f); //linear_rep_limit_close
498  else strength = (-1)*expf((-1)* GRAVITY * logf(d)); //exp_rep_limit_mid_far
499  force = scale_up_points(&force,(strength/magnitude));
500  return force;
501 }
