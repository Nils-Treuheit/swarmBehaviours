749 static Point log_Attraction(Point* pos, Point* target, float cutOff)

768 static Point linear_Attraction(Point* pos, Point* target, float max_val)
769 {
770  Point force = sub_points(target, pos);

772  float d = magnitude; //fminf(magnitude, limit);

787 static Point limExp_Repulsion(Point* pos, Point* target, float cutOff, float max_val)
788 {
789  Point force = sub_points(target, pos);
790  float magnitude = mag(&force);
791  float strength;
792  float d = magnitude; //fminf(magnitude, limit);
793  if (d<cutOff) strength = (-1)*(GRAVITY * max_val)/fmaxf(d,1.0f); //linear_rep_limit_close
794  else strength = (-1)*expf((-1)* GRAVITY * logf(d)); //exp_rep_limit_mid_far
795  force = scale_up_points(&force,(strength/magnitude));
796  return force;
797 }

849  + Context Map Preparation +
850  +++++++++++++++++++++++++++

854  */

873  amd the maximum of those are assigned. */

894  max_member_atts[idx] = strongest_force((max_member_atts+idx),&att_force);

897  max_member_reps[idx] = strongest_force((max_member_reps+idx), &rep_force);
898  }

907  {

909  dist = sub_points(&danger,&pos);
910  //rotated = rotate2D(&dist,M_PI);

912  {

914 
915  rep_force = limExp_Repulsion(&pos,&danger,DANGER_CUT_OFF,DANGER_MAX);
916  max_danger_forces[idx] = strongest_force((max_danger_forces+idx),&rep_force);
917  }
918  }
919  }

923  ++++++++++++++++++++++++++
924  + Context Map Evaluation +
925  ++++++++++++++++++++++++++
926 
927  evaluate context steering behavior
928  */
929 
930  /* calculate scores with respect to context map weightings and swarm alignment */
931  if(alignment_counter>0.0f) alignment_force = scale_down_points(&alignment_force,alignment_counter);
932  for(int idx=0; idx<NUM_DIRECTION_RAYS; ++idx)
933  {
934  score[idx] = 0.000f;
935  score[idx] = mag((max_interest_forces+idx))*GOAL_MULT;
936  score[idx] = score[idx]+mag((max_member_atts+idx))*DRONE_ATT_MULT;
937  score[idx] = score[idx]-mag((max_member_reps+idx))*DRONE_REP_MULT;
938  score[idx] = score[idx]-mag((max_danger_forces+idx))*DANGER_MULT;
939 
940  /* more likely to perform alignment otherwise less likely to switch directions */
941  alignSim = cosine_sim((DIRECTION_RAYS+idx),&alignment_force);
942  velSim = cosine_sim((DIRECTION_RAYS+idx),&vel);
943  alignmentMap[idx] = alignSim * (mag(&alignment_force)/MAX_SPEED) * ALIGN_MULT;
944  selfAlignmentMap[idx] = velSim * (mag(&vel)/MAX_SPEED) * SELF_ALIGN_MULT;
945  if(mag(&alignment_force)>0.01f) score[idx] = score[idx] + alignmentMap[idx];
946  if(mag(&vel)>0.01f) score[idx] = score[idx] + selfAlignmentMap[idx];
947  }
948 
949 
950  /*select strongest force as main force direction */
951  while((maxIdx < NUM_DIRECTION_RAYS) && !mask[maxIdx]) maxIdx+=1;
952  for(int idx=maxIdx+1; idx<NUM_DIRECTION_RAYS; ++idx)
953  if(mask[idx] && (score[idx] > score[maxIdx])) maxIdx = idx;
954 
955  if(maxIdx < NUM_DIRECTION_RAYS)
956  {
957  if(fabsf(score[maxIdx])>0.002f)
958  {
959  /*interpolate between main and strongest neighboring force */
960  leftIdx = (maxIdx-1 + NUM_DIRECTION_RAYS)%NUM_DIRECTION_RAYS;
961  rightIdx = (maxIdx+1 + NUM_DIRECTION_RAYS)%NUM_DIRECTION_RAYS;
962  neighborIdx = -1;
963  if(score[leftIdx]<score[rightIdx])
964  {
965  if(mask[rightIdx]) neighborIdx = rightIdx;
966  else if(mask[leftIdx]) neighborIdx = leftIdx;
967  }
968  else
969  {
970  if(mask[leftIdx]) neighborIdx = leftIdx;
971  else if(mask[rightIdx]) neighborIdx = rightIdx;
972  }
973  if(neighborIdx>=0)
974  {
975  magnitude = score[maxIdx]/(score[maxIdx]+score[neighborIdx]);
976  resulting_force = scale_up_points((DIRECTION_RAYS+maxIdx),magnitude);
977  scaled_force = scale_up_points((DIRECTION_RAYS+neighborIdx),(1.0f-magnitude));
978  resulting_force = add_points(&resulting_force,&scaled_force);
979  }
980  }
981  else resulting_force = scale_up_points((DIRECTION_RAYS+maxIdx),1.0f);
982  }
983  else scale_up_points(&vel,0.0f);
984 
985  if(fabsf(score[maxIdx])<0.0001f)
986  {
987  scale_up_points(&vel,0.0f);
988  scale_up_points(&acc,0.0f);
989  }
990  else
991  {
992  /*set drone attributes accelleration, velocity, position */
993  acc = scale_up_points(&resulting_force,(MAX_SPEED/mag(&resulting_force)));
994  }
995

997  velocity->x = MAX_SPEED * tanhf(vel.x/MAX_SPEED+acc.x/MAX_SPEED);
998  velocity->y = MAX_SPEED * tanhf(vel.y/MAX_SPEED+acc.y/MAX_SPEED);
999  acInfoSetVelocityEnu_f(AC_ID,velocity);
1000 
1001  struct EnuCoor_i future_pos = *stateGetPositionEnu_i();
1002  future_pos.x += POS_BFP_OF_REAL(velocity->x);
1003  future_pos.y += POS_BFP_OF_REAL(velocity->y);
1004  future_pos.z = POS_BFP_OF_REAL(FLIGHT_HEIGHT);
1005 
1006  /* Move the waypoints */
1007  current_pos = *stateGetPositionLla_f();
1008  waypoint_set_enu_i(SWARM_WAYPOINT_ID, &future_pos);
1009  for(uint8_t wp_id = FIRST_GOAL_POINT_ID; wp_id < LAST_GOAL_POINT_ID; ++wp_id)
1010  {
1011  struct LlaCoor_i* way_point = waypoint_get_lla(wp_id);
1012  att_point = toFloatPointFormat(way_point);
1013  updateSyncLinkMsg(&current_pos, wp_id);
1014  }
1015

1017  {
1018  tick = 3;
1019  write_to_log_file();

1021  alignmentMap, selfAlignmentMap, mask, alignment_force, maxIdx, neighborIdx);
1022  write_to_swarm_info_debug_file(other_acs,AC_AMOUNT);
1023  }
